
Задание 1
SELECT Стаж, Место прописки
FROM Водители
WHERE Место LIKE ‘eee’ AND (Стаж !=3 and Стаж !=5 and Стаж !=6 and Стаж !=10) 

Задание 2
SELECT D.№ Паспорта
FROM Диспетчерская as V
            LEFT JOIN Водители  as D 
ON V.Код водителя= D.Код водителя
WHERE D.Время отбытия IS NULL OR D.Путевка IS NULL

Задание 3
SELECT А.Регистрационный № авто, D.Время прибытия, D.Время отбытия, M.Пункт назначения, M.Пункт назначения
FROM Диспетчерская as D 
            LEFT JOIN Автомобили as А 
ON D.Код авто=А.Код авто
            LEFT JOIN Маршрут as M 
ON D.Путевка=М.Код маршрута
WHERE A.Год выпуска авто = 2015
